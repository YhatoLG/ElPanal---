CREATE DEFINER=`root`@`localhost` PROCEDURE `ModificarPais`(
	IN `P_IdPais` CHAR(36),
	IN `P_Nombre` VARCHAR(255)
)
LANGUAGE SQL
NOT DETERMINISTIC
CONTAINS SQL
SQL SECURITY DEFINER
COMMENT ''
BEGIN
UPDATE Pais
	SET Nombre = P_Nombre,
			Actualiza = CURRENT_TIMESTAMP()
	WHERE IdPais = P_IdPais
		AND Activo = 1;
END
